# Plaidstone Webhook Dispatcher — v2

## `README.md`

```markdown
# Plaidstone Webhook Dispatcher

Async service that consumes `SettlementEvent` messages from a Redis stream and
dispatches signed HTTP POSTs to merchant webhook URLs. Built to replace the
duplicate-batch behavior seen during the March 3, 2026 outage (PLD-2147).

## Features

- Per-merchant token-bucket rate limiting (Redis `INCR`/`EXPIRE`, survives restarts).
- HMAC-SHA256 signing in Stripe-style `t=<ts>,v1=<hex>` format.
- Exponential backoff retries (base 2s, cap 60s, max 5 attempts) with jitter.
- Idempotency-key dedup via Redis SET with 7-day TTL.
- Per-merchant circuit breaker (opens after 5 consecutive 5xx, half-open after 30s).
- Prometheus metrics on `:9108/metrics`.
- Graceful SIGTERM shutdown.
- `structlog` JSON logs.

## `dispatcher/signing.py`

```python
"""HMAC signing + timestamp verification (Stripe-style)."""
from __future__ import annotations

import hmac
import time
from hashlib import sha256
from typing import Tuple

MAX_SKEW_SECONDS = 300  # reject events older than 5 min


def sign(secret: str, body: bytes, ts: int | None = None) -> Tuple[str, int]:
    ts = ts if ts is not None else int(time.time())
    signed_payload = f"{ts}.".encode() + body
    digest = hmac.new(secret.encode(), signed_payload, sha256).hexdigest()
    return f"t={ts},v1={digest}", ts


def verify(secret: str, body: bytes, header: str, now: int | None = None) -> bool:
    now = now if now is not None else int(time.time())
    parts = dict(p.split("=", 1) for p in header.split(",") if "=" in p)
    try:
        ts = int(parts["t"])
        provided = parts["v1"]
    except (KeyError, ValueError):
        return False
    if abs(now - ts) > MAX_SKEW_SECONDS:
        return False
    signed_payload = f"{ts}.".encode() + body
    expected = hmac.new(secret.encode(), signed_payload, sha256).hexdigest()
    return hmac.compare_digest(expected, provided)
```

## `dispatcher/retry.py`

```python
"""Retry policy + DLQ push."""
from __future__ import annotations

import json
import random
from dataclasses import asdict, dataclass
from typing import Optional

import redis.asyncio as redis

from events.models import SettlementEvent

DLQ_STREAM = "settlements:dlq"


@dataclass
class RetryPolicy:
    base_seconds: float = 2.0
    cap_seconds: float = 60.0
    max_attempts: int = 5

    def delay(self, attempt: int) -> float:
        """Return jittered delay for a 1-indexed attempt number."""
        if attempt < 1:
            return 0.0
        raw = self.base_seconds * (2 ** (attempt - 1))
        capped = min(raw, self.cap_seconds)
        # full jitter: [0, capped]
        return random.uniform(0, capped)

    def should_retry(self, attempt: int) -> bool:
        return attempt < self.max_attempts


async def push_dlq(
    client: redis.Redis,
    event: SettlementEvent,
    reason: str,
    last_status: Optional[int] = None,
) -> None:
    payload = asdict(event)
    payload["settled_at"] = event.settled_at.isoformat()
    await client.xadd(
        DLQ_STREAM,
        {
            "event": json.dumps(payload),
            "reason": reason,
            "last_status": str(last_status) if last_status is not None else "",
        },
    )
```

## `dispatcher/ratelimit.py`

```python
"""Redis-backed token-bucket-ish rate limiter.

Uses the fixed-window `INCR`+`EXPIRE` pattern per-merchant. This survives
restarts because state lives in Redis. Under contention, callers await until
the current second's budget has room.
"""
from __future__ import annotations

import asyncio
import time

import redis.asyncio as redis


class RedisRateLimiter:
    def __init__(self, client: redis.Redis):
        self._client = client

    async def acquire(self, merchant_id: str, max_rps: int) -> None:
        while True:
            now = int(time.time())
            key = f"rl:{merchant_id}:{now}"
            count = await self._client.incr(key)
            if count == 1:
                await self._client.expire(key, 2)
            if count <= max_rps:
                return
            # wait until the next second window
            sleep_for = (now + 1) - time.time()
            if sleep_for > 0:
                await asyncio.sleep(sleep_for)
```

## `dispatcher/breaker.py`

```python
"""Per-merchant circuit breaker."""
from __future__ import annotations

import time
from dataclasses import dataclass
from enum import Enum


class State(str, Enum):
    CLOSED = "closed"
    OPEN = "open"
    HALF_OPEN = "half_open"


@dataclass
class Breaker:
    consecutive_5xx: int = 0
    opened_at: float = 0.0
    state: State = State.CLOSED

    threshold: int = 5
    cooldown_s: float = 30.0

    def allow(self, now: float | None = None) -> bool:
        now = now if now is not None else time.time()
        if self.state is State.OPEN:
            if now - self.opened_at >= self.cooldown_s:
                self.state = State.HALF_OPEN
                return True
            return False
        return True

    def on_success(self) -> None:
        self.consecutive_5xx = 0
        self.state = State.CLOSED

    def on_5xx(self, now: float | None = None) -> None:
        now = now if now is not None else time.time()
        self.consecutive_5xx += 1
        if self.consecutive_5xx >= self.threshold:
            self.state = State.OPEN
            self.opened_at = now
```

## `dispatcher/client.py`

```python
"""WebhookDispatcher main async loop."""
from __future__ import annotations

import asyncio
import json
import os
import signal
from dataclasses import asdict
from datetime import datetime
from typing import Dict, Optional

import asyncpg
import httpx
import redis.asyncio as redis
import structlog

from events.models import SettlementEvent

from .breaker import Breaker
from .metrics import (
    start as start_metrics,
    webhook_latency_seconds,
    webhook_retries_total,
    webhooks_sent_total,
)
from .ratelimit import RedisRateLimiter
from .retry import RetryPolicy, push_dlq
from .signing import sign

STREAM = "settlements:v2"
GROUP = "dispatcher-g1"
CONSUMER = "c1"
DEDUP_TTL_SECONDS = 7 * 24 * 3600

structlog.configure(
    processors=[
        structlog.processors.TimeStamper(fmt="iso"),
        structlog.processors.add_log_level,
        structlog.processors.JSONRenderer(),
    ]
)
log = structlog.get_logger("dispatcher")


class WebhookDispatcher:
    def __init__(
        self,
        redis_client: redis.Redis,
        db_pool: asyncpg.Pool,
        http_client: httpx.AsyncClient,
        concurrency: int = 32,
        retry_policy: Optional[RetryPolicy] = None,
    ) -> None:
        self._redis = redis_client
        self._db = db_pool
        self._http = http_client
        self._concurrency = concurrency
        self._sem = asyncio.Semaphore(concurrency)
        self._limiter = RedisRateLimiter(redis_client)
        self._retry = retry_policy or RetryPolicy()
        self._breakers: Dict[str, Breaker] = {}
        self._stop = asyncio.Event()

    # --- lifecycle ------------------------------------------------------
    async def ensure_group(self) -> None:
        try:
            await self._redis.xgroup_create(STREAM, GROUP, id="0", mkstream=True)
        except redis.ResponseError as e:
            if "BUSYGROUP" not in str(e):
                raise

    def request_stop(self) -> None:
        self._stop.set()

    # --- helpers --------------------------------------------------------
    def _breaker(self, merchant_id: str) -> Breaker:
        b = self._breakers.get(merchant_id)
        if b is None:
            b = Breaker()
            self._breakers[merchant_id] = b
        return b

    async def _lookup(self, merchant_id: str) -> dict | None:
        row = await self._db.fetchrow(
            "SELECT merchant_id, url, secret, max_rps, active "
            "FROM merchant_webhooks WHERE merchant_id = $1",
            merchant_id,
        )
        return dict(row) if row else None

    async def _already_delivered(self, idem_key: str) -> bool:
        added = await self._redis.set(
            f"idem:{idem_key}", "1", nx=True, ex=DEDUP_TTL_SECONDS
        )
        return added is None or added is False

    @staticmethod
    def _serialize(event: SettlementEvent) -> bytes:
        payload = asdict(event)
        payload["settled_at"] = event.settled_at.isoformat()
        return json.dumps(payload, separators=(",", ":")).encode()

    @staticmethod
    def _parse(fields: Dict[str, str]) -> SettlementEvent:
        return SettlementEvent(
            event_id=fields["event_id"],
            merchant_id=fields["merchant_id"],
            amount_cents=int(fields["amount_cents"]),
            currency=fields["currency"],
            settled_at=datetime.fromisoformat(fields["settled_at"]),
            idempotency_key=fields["idempotency_key"],
        )

    # --- delivery -------------------------------------------------------
    async def _post_once(
        self, url: str, body: bytes, signature: str, ts: int
    ) -> httpx.Response:
        headers = {
            "Content-Type": "application/json",
            "X-Plaidstone-Signature": signature,
            "X-Plaidstone-Timestamp": str(ts),
        }
        return await self._http.post(url, content=body, headers=headers)

    async def deliver(self, event: SettlementEvent) -> None:
        merchant = await self._lookup(event.merchant_id)
        if not merchant or not merchant["active"]:
            log.info("skip_inactive", merchant_id=event.merchant_id)
            return

        if await self._already_delivered(event.idempotency_key):
            log.info("duplicate_dropped", idem=event.idempotency_key)
            webhooks_sent_total.labels(event.merchant_id, "duplicate").inc()
            return

        breaker = self._breaker(event.merchant_id)
        if not breaker.allow():
            log.warning("breaker_open", merchant_id=event.merchant_id)
            webhooks_sent_total.labels(event.merchant_id, "breaker_open").inc()
            await push_dlq(self._redis, event, "breaker_open")
            return

        body = self._serialize(event)
        attempt = 0
        last_status: Optional[int] = None
        while True:
            attempt += 1
            await self._limiter.acquire(event.merchant_id, int(merchant["max_rps"]))
            signature, ts = sign(merchant["secret"], body)
            try:
                with webhook_latency_seconds.labels(event.merchant_id).time():
                    r = await self._post_once(merchant["url"], body, signature, ts)
                last_status = r.status_code
                if 200 <= r.status_code < 300:
                    breaker.on_success()
                    webhooks_sent_total.labels(event.merchant_id, "ok").inc()
                    return
                if 500 <= r.status_code < 600:
                    breaker.on_5xx()
                    outcome = "5xx"
                else:
                    outcome = "4xx"
            except httpx.HTTPError:
                outcome = "error"
                last_status = None

            webhook_retries_total.labels(event.merchant_id, outcome).inc()

            if not self._retry.should_retry(attempt):
                await push_dlq(self._redis, event, outcome, last_status)
                webhooks_sent_total.labels(event.merchant_id, "dlq").inc()
                return

            await asyncio.sleep(self._retry.delay(attempt))

    # --- run loop -------------------------------------------------------
    async def run(self) -> None:
        await self.ensure_group()
        while not self._stop.is_set():
            try:
                resp = await self._redis.xreadgroup(
                    GROUP, CONSUMER, {STREAM: ">"}, count=32, block=1000
                )
            except redis.ConnectionError:
                await asyncio.sleep(1.0)
                continue
            if not resp:
                continue
            coros = []
            msg_ids = []
            for _stream, messages in resp:
                for msg_id, fields in messages:
                    msg_ids.append(msg_id)
                    event = self._parse(fields)
                    coros.append(self._guarded(event))
            if coros:
                await asyncio.gather(*coros, return_exceptions=True)
                await self._redis.xack(STREAM, GROUP, *msg_ids)

    async def _guarded(self, event: SettlementEvent) -> None:
        async with self._sem:
            try:
                await self.deliver(event)
            except Exception as e:  # keep loop alive
                log.error("deliver_failed", event_id=event.event_id, error=str(e))


async def _main() -> None:
    redis_url = os.environ.get("REDIS_URL", "redis://redis:6379/0")
    db_url = os.environ.get("DB_URL", "postgresql://plaidstone@db/plaidstone")
    concurrency = int(os.environ.get("CONCURRENCY", "32"))
    metrics_port = int(os.environ.get("METRICS_PORT", "9108"))

    start_metrics(metrics_port)
    r = redis.from_url(redis_url, decode_responses=True)
    pool = await asyncpg.create_pool(db_url, min_size=1, max_size=8)
    async with httpx.AsyncClient(timeout=10.0) as http:
        dispatcher = WebhookDispatcher(r, pool, http, concurrency=concurrency)

        loop = asyncio.get_running_loop()
        for s in (signal.SIGTERM, signal.SIGINT):
            loop.add_signal_handler(s, dispatcher.request_stop)

        try:
            await dispatcher.run()
        finally:
            await pool.close()
            await r.close()


if __name__ == "__main__":
    asyncio.run(_main())
```

"""Dispatcher tests: signing, dedup, rate-limit, retries, breaker."""
from __future__ import annotations

import asyncio
import time
from datetime import datetime, timezone

import httpx
import pytest
import respx

from dispatcher.client import WebhookDispatcher
from dispatcher.retry import RetryPolicy
from dispatcher.signing import sign, verify
from events.models import SettlementEvent


def _event(i: int = 1, idem: str | None = None) -> SettlementEvent:
    return SettlementEvent(
        event_id=f"evt_{i}",
        merchant_id="merch_8f2a",
        amount_cents=1000 + i,
        currency="USD",
        settled_at=datetime(2026, 4, 19, tzinfo=timezone.utc),
        idempotency_key=idem or f"idem-{i}",
    )


def test_sign_and_verify_roundtrip():
    header, ts = sign("sekrit", b"{\"x\":1}")
    assert verify("sekrit", b"{\"x\":1}", header, now=ts)


def test_verify_rejects_old_timestamp():
    header, ts = sign("sekrit", b"{}")
    assert not verify("sekrit", b"{}", header, now=ts + 3600)


def test_verify_rejects_tampered_body():
    header, ts = sign("sekrit", b"{\"a\":1}")
    assert not verify("sekrit", b"{\"a\":2}", header, now=ts)
