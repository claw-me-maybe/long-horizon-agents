## `README.md`

```markdown
# Plaidstone Webhook Dispatcher (v1)

Minimal async dispatcher that reads `SettlementEvent` messages from the Redis
stream `settlements:v2` (consumer group `dispatcher-g1`) and POSTs them to
merchant webhook URLs registered in the `merchant_webhooks` Postgres table.

## Env vars

- `REDIS_URL` (default `redis://localhost:6379/0`)
- `DB_URL` (default `postgresql://plaidstone@localhost/plaidstone`)
- `CONCURRENCY` (default `16`)

## Run

```
pip install -r requirements.txt
python -m dispatcher.client
```

This v1 is a minimal happy-path; retries/signing/metrics are pending.
```

## `dispatcher/client.py`

```python
"""Minimal async webhook dispatcher (v1).

Reads settlement events from a Redis stream and POSTs them to the merchant
webhook URL registered in Postgres. Uses a per-merchant asyncio.Semaphore to
crudely cap concurrency.
"""
from __future__ import annotations

import asyncio
import json
import logging
import os
from dataclasses import asdict
from datetime import datetime
from typing import Dict

import asyncpg
import httpx
import redis.asyncio as redis

from events.models import SettlementEvent

log = logging.getLogger("dispatcher")
logging.basicConfig(level=logging.INFO)

STREAM = "settlements:v2"
GROUP = "dispatcher-g1"
CONSUMER = "c1"


class WebhookDispatcher:
    def __init__(
        self,
        redis_url: str,
        db_url: str,
        concurrency: int = 16,
    ) -> None:
        self.redis_url = redis_url
        self.db_url = db_url
        self.concurrency = concurrency
        self._sem: Dict[str, asyncio.Semaphore] = {}
        self._redis: redis.Redis | None = None
        self._db: asyncpg.Pool | None = None
        self._http: httpx.AsyncClient | None = None

    async def start(self) -> None:
        self._redis = redis.from_url(self.redis_url, decode_responses=True)
        self._db = await asyncpg.create_pool(self.db_url, min_size=1, max_size=4)
        self._http = httpx.AsyncClient(timeout=10.0)
        try:
            await self._redis.xgroup_create(STREAM, GROUP, id="0", mkstream=True)
        except redis.ResponseError:
            pass  # group already exists

    async def close(self) -> None:
        if self._http:
            await self._http.aclose()
        if self._db:
            await self._db.close()
        if self._redis:
            await self._redis.close()

    def _sem_for(self, merchant_id: str) -> asyncio.Semaphore:
        sem = self._sem.get(merchant_id)
        if sem is None:
            sem = asyncio.Semaphore(self.concurrency)
            self._sem[merchant_id] = sem
        return sem

    async def _lookup_merchant(self, merchant_id: str) -> dict | None:
        assert self._db
        row = await self._db.fetchrow(
            "SELECT merchant_id, url, secret, max_rps, active "
            "FROM merchant_webhooks WHERE merchant_id = $1",
            merchant_id,
        )
        return dict(row) if row else None

    async def _deliver(self, event: SettlementEvent) -> None:
        merchant = await self._lookup_merchant(event.merchant_id)
        if not merchant or not merchant["active"]:
            log.info("skip inactive merchant=%s", event.merchant_id)
            return
        payload = asdict(event)
        payload["settled_at"] = event.settled_at.isoformat()
        body = json.dumps(payload).encode()
        assert self._http
        async with self._sem_for(event.merchant_id):
            r = await self._http.post(merchant["url"], content=body)
            log.info("delivered event=%s status=%s", event.event_id, r.status_code)

    async def _parse(self, fields: Dict[str, str]) -> SettlementEvent:
        return SettlementEvent(
            event_id=fields["event_id"],
            merchant_id=fields["merchant_id"],
            amount_cents=int(fields["amount_cents"]),
            currency=fields["currency"],
            settled_at=datetime.fromisoformat(fields["settled_at"]),
            idempotency_key=fields["idempotency_key"],
        )

    async def run(self, stop_event: asyncio.Event | None = None) -> None:
        assert self._redis
        stop_event = stop_event or asyncio.Event()
        while not stop_event.is_set():
            resp = await self._redis.xreadgroup(
                GROUP, CONSUMER, {STREAM: ">"}, count=16, block=1000
            )
            if not resp:
                continue
            tasks = []
            msg_ids = []
            for _stream, messages in resp:
                for msg_id, fields in messages:
                    event = await self._parse(fields)
                    tasks.append(self._deliver(event))
                    msg_ids.append(msg_id)
            await asyncio.gather(*tasks, return_exceptions=True)
            if msg_ids:
                await self._redis.xack(STREAM, GROUP, *msg_ids)


async def _main() -> None:
    d = WebhookDispatcher(
        redis_url=os.environ.get("REDIS_URL", "redis://localhost:6379/0"),
        db_url=os.environ.get("DB_URL", "postgresql://plaidstone@localhost/plaidstone"),
        concurrency=int(os.environ.get("CONCURRENCY", "16")),
    )
    await d.start()
    try:
        await d.run()
    finally:
        await d.close()


if __name__ == "__main__":
    asyncio.run(_main())
```

## `tests/test_dispatcher.py`

```python
"""Happy-path test for v1 dispatcher."""
import asyncio
import json
from datetime import datetime, timezone
from unittest.mock import AsyncMock, MagicMock

import httpx
import pytest
import respx

from dispatcher.client import WebhookDispatcher
from events.models import SettlementEvent


@pytest.mark.asyncio
@respx.mock
async def test_deliver_happy_path():
    route = respx.post("https://hooks.acme-retail.com/plaidstone").mock(
        return_value=httpx.Response(200)
    )

    d = WebhookDispatcher(
        redis_url="redis://localhost:6379/0",
        db_url="postgresql://ignored",
    )
    d._http = httpx.AsyncClient()
    d._db = MagicMock()
    d._db.fetchrow = AsyncMock(
        return_value={
            "merchant_id": "merch_8f2a",
            "url": "https://hooks.acme-retail.com/plaidstone",
            "secret": "whsec_test",
            "max_rps": 5,
            "active": True,
        }
    )

    event = SettlementEvent(
        event_id="evt_1",
        merchant_id="merch_8f2a",
        amount_cents=1000,
        currency="USD",
        settled_at=datetime(2026, 4, 19, tzinfo=timezone.utc),
        idempotency_key="idem-1",
    )
    await d._deliver(event)

    assert route.called
    body = json.loads(route.calls.last.request.content)
    assert body["event_id"] == "evt_1"
    assert body["merchant_id"] == "merch_8f2a"
    await d._http.aclose()
```
