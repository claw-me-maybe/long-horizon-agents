## `README.md`

# plaidstone-webhook-dispatcher

Async webhook dispatcher for Plaidstone Financial settlement events (ticket PLD-2147).
Consumes from Redis stream `settlements:v2` (consumer group `dispatcher-g1`) and
signs + POSTs each event to the merchant's registered URL.

## Features
- Per-merchant token-bucket rate limiter backed by Redis (survives restarts)
- HMAC-SHA256 signing, Stripe-style `t=<ts>,v1=<hex>`
- Idempotency-key dedup in Redis SET (7-day TTL) — at-least-once with dedup
- Jittered exponential backoff, dead-letter queue `settlements:dlq`
- Per-merchant circuit breaker (5 consecutive 5xx -> open, 30s half-open)
- Prometheus metrics on :9100 (`webhooks_sent_total`, `webhook_latency_seconds`,
  `webhook_retries_total`)
- `structlog` JSON logs, graceful SIGTERM shutdown

## `dispatcher/signing.py`

```python
"""HMAC-SHA256 signing + timestamp-tolerant verification.

Stripe-style header: `t=<unix_ts>,v1=<hex>`.
"""
from __future__ import annotations

import hashlib
import hmac
import time


MAX_AGE_SECONDS = 300  # reject anything older than 5 minutes


def sign(secret: str, body: bytes, timestamp: int | None = None) -> tuple[str, str]:
    """Return (timestamp_str, header_value)."""
    ts = timestamp if timestamp is not None else int(time.time())
    payload = f"{ts}.".encode() + body
    digest = hmac.new(secret.encode(), payload, hashlib.sha256).hexdigest()
    return str(ts), f"t={ts},v1={digest}"


def _parse_header(header: str) -> tuple[int, str]:
    parts = dict(p.split("=", 1) for p in header.split(","))
    return int(parts["t"]), parts["v1"]


def verify(
    secret: str,
    body: bytes,
    header: str,
    *,
    now: int | None = None,
    max_age: int = MAX_AGE_SECONDS,
) -> bool:
    """Constant-time HMAC verification + freshness check."""
    try:
        ts, got = _parse_header(header)
    except (KeyError, ValueError):
        return False
    current = now if now is not None else int(time.time())
    if abs(current - ts) > max_age:
        return False
    payload = f"{ts}.".encode() + body
    expected = hmac.new(secret.encode(), payload, hashlib.sha256).hexdigest()
    return hmac.compare_digest(expected, got)
```

## `dispatcher/retry.py`

```python
"""Retry policy: jittered exponential backoff + dead-letter queue."""
from __future__ import annotations

import json
import random
from dataclasses import dataclass

import redis.asyncio as aioredis


DLQ_STREAM = "settlements:dlq"


@dataclass(frozen=True)
class RetryPolicy:
    base: float = 2.0
    cap: float = 60.0
    max_attempts: int = 5

    def backoff(self, attempt: int) -> float:
        """attempt is 1-based. Full jitter over [0, min(cap, base * 2^(attempt-1))]."""
        window = min(self.cap, self.base * (2 ** (attempt - 1)))
        return random.uniform(0, window)

    def exhausted(self, attempt: int) -> bool:
        return attempt >= self.max_attempts


async def dead_letter(
    redis: aioredis.Redis, event_id: str, merchant_id: str, reason: str
) -> None:
    await redis.xadd(
        DLQ_STREAM,
        {
            "event_id": event_id,
            "merchant_id": merchant_id,
            "reason": reason,
            "payload": json.dumps({"event_id": event_id, "merchant_id": merchant_id}),
        },
    )
```

## `dispatcher/client.py`

```python
"""Rate-limited, signed, retrying webhook dispatcher."""
from __future__ import annotations

import asyncio
import json
import os
import signal
import time
from contextlib import suppress
from dataclasses import dataclass, field
from datetime import datetime
from typing import Dict

import asyncpg
import httpx
import redis.asyncio as aioredis
import structlog
from prometheus_client import Counter, Histogram, start_http_server

from dispatcher.retry import RetryPolicy, dead_letter
from dispatcher.signing import sign
from events.models import SettlementEvent

structlog.configure(
    processors=[
        structlog.processors.add_log_level,
        structlog.processors.TimeStamper(fmt="iso"),
        structlog.processors.JSONRenderer(),
    ],
)
log = structlog.get_logger()

STREAM = "settlements:v2"
GROUP = "dispatcher-g1"
CONSUMER = os.environ.get("CONSUMER_ID", "worker-1")
DEDUP_SET_PREFIX = "dispatcher:seen:"
DEDUP_TTL_SECONDS = 7 * 24 * 3600

SENT = Counter(
    "webhooks_sent_total", "Webhooks sent", ["merchant_id", "outcome"]
)
LATENCY = Histogram(
    "webhook_latency_seconds", "Webhook POST latency", ["merchant_id"]
)
RETRIES = Counter(
    "webhook_retries_total", "Retries", ["merchant_id", "outcome"]
)


@dataclass
class CircuitBreaker:
    failure_threshold: int = 5
    reset_after: float = 30.0
    consecutive_failures: int = 0
    opened_at: float | None = None

    def record_success(self) -> None:
        self.consecutive_failures = 0
        self.opened_at = None

    def record_failure(self) -> None:
        self.consecutive_failures += 1
        if self.consecutive_failures >= self.failure_threshold and self.opened_at is None:
            self.opened_at = time.monotonic()

    def allow(self) -> bool:
        if self.opened_at is None:
            return True
        # half-open after reset_after: let a single probe through
        if time.monotonic() - self.opened_at >= self.reset_after:
            return True
        return False


class TokenBucket:
    """Redis-backed fixed-window counter that survives restarts.

    Uses INCR + EXPIRE on a bucket keyed by the current integer second.
    Simple, correct, and cheap; trades some burstiness at second boundaries
    for durability and cluster-wide coordination.
    """

    def __init__(self, redis: aioredis.Redis) -> None:
        self._redis = redis

    async def acquire(self, merchant_id: str, max_rps: int) -> None:
        while True:
            now = int(time.time())
            key = f"dispatcher:rl:{merchant_id}:{now}"
            count = await self._redis.incr(key)
            if count == 1:
                await self._redis.expire(key, 2)
            if count <= max_rps:
                return
            # wait until the next second boundary
            await asyncio.sleep(max(0.005, 1.0 - (time.time() - now)))


class WebhookDispatcher:
    def __init__(
        self,
        redis_url: str,
        db_url: str,
        concurrency: int = 32,
        retry_policy: RetryPolicy | None = None,
        *,
        redis: aioredis.Redis | None = None,
        db: asyncpg.Pool | None = None,
        http: httpx.AsyncClient | None = None,
    ) -> None:
        self.redis_url = redis_url
        self.db_url = db_url
        self.concurrency = concurrency
        self.retry_policy = retry_policy or RetryPolicy()
        self._redis = redis
        self._db = db
        self._http = http
        self._bucket: TokenBucket | None = None
        self._breakers: Dict[str, CircuitBreaker] = {}
        self._merchant_cache: Dict[str, dict] = {}
        self._stop = asyncio.Event()

    async def start(self) -> None:
        if self._redis is None:
            self._redis = aioredis.from_url(self.redis_url, decode_responses=True)
        if self._db is None:
            self._db = await asyncpg.create_pool(self.db_url, min_size=1, max_size=8)
        if self._http is None:
            self._http = httpx.AsyncClient(timeout=10.0)
        self._bucket = TokenBucket(self._redis)
        with suppress(Exception):
            await self._redis.xgroup_create(STREAM, GROUP, id="0", mkstream=True)

    async def stop(self) -> None:
        self._stop.set()
        if self._http:
            await self._http.aclose()
        if self._db:
            await self._db.close()
        if self._redis:
            await self._redis.aclose()

    async def _merchant(self, merchant_id: str) -> dict | None:
        if merchant_id in self._merchant_cache:
            return self._merchant_cache[merchant_id]
        assert self._db is not None
        row = await self._db.fetchrow(
            "SELECT merchant_id, url, secret, max_rps, active "
            "FROM merchant_webhooks WHERE merchant_id = $1",
            merchant_id,
        )
        if not row:
            return None
        data = dict(row)
        self._merchant_cache[merchant_id] = data
        return data

    def _breaker(self, merchant_id: str) -> CircuitBreaker:
        cb = self._breakers.get(merchant_id)
        if cb is None:
            cb = CircuitBreaker()
            self._breakers[merchant_id] = cb
        return cb

    async def _already_seen(self, idem_key: str) -> bool:
        assert self._redis is not None
        key = DEDUP_SET_PREFIX + idem_key
        # SET NX with EX = atomic "claim this key or confirm it exists"
        ok = await self._redis.set(key, "1", nx=True, ex=DEDUP_TTL_SECONDS)
        return not bool(ok)

    async def _post_once(
        self, event: SettlementEvent, merchant: dict, body: bytes
    ) -> httpx.Response:
        ts, sig = sign(merchant["secret"], body)
        headers = {
            "Content-Type": "application/json",
            "X-Plaidstone-Timestamp": ts,
            "X-Plaidstone-Signature": sig,
        }
        assert self._http is not None
        with LATENCY.labels(merchant_id=event.merchant_id).time():
            return await self._http.post(merchant["url"], content=body, headers=headers)

    async def _dispatch(self, msg_id: str, event: SettlementEvent) -> None:
        merchant = await self._merchant(event.merchant_id)
        if not merchant or not merchant["active"]:
            await self._ack(msg_id)
            return

        if await self._already_seen(event.idempotency_key):
            log.info("dedup", event_id=event.event_id, merchant_id=event.merchant_id)
            SENT.labels(merchant_id=event.merchant_id, outcome="dedup").inc()
            await self._ack(msg_id)
            return

        breaker = self._breaker(event.merchant_id)
        if not breaker.allow():
            SENT.labels(merchant_id=event.merchant_id, outcome="circuit_open").inc()
            # do NOT ack: re-delivery will retry once breaker half-opens
            return

        body = json.dumps(
            {
                "event_id": event.event_id,
                "merchant_id": event.merchant_id,
                "amount_cents": event.amount_cents,
                "currency": event.currency,
                "settled_at": event.settled_at.isoformat(),
                "idempotency_key": event.idempotency_key,
            }
        ).encode()

        attempt = 0
        while True:
            attempt += 1
            assert self._bucket is not None
            await self._bucket.acquire(event.merchant_id, merchant["max_rps"])
            try:
                resp = await self._post_once(event, merchant, body)
            except (httpx.RequestError, asyncio.TimeoutError) as exc:
                status = None
                log.warning(
                    "post_error",
                    event_id=event.event_id,
                    merchant_id=event.merchant_id,
                    attempt=attempt,
                    error=str(exc),
                )
            else:
                status = resp.status_code

            if status is not None and 200 <= status < 300:
                breaker.record_success()
                SENT.labels(merchant_id=event.merchant_id, outcome="ok").inc()
                await self._ack(msg_id)
                return

            is_5xx = status is not None and 500 <= status < 600
            if is_5xx or status is None:
                breaker.record_failure()
            RETRIES.labels(
                merchant_id=event.merchant_id,
                outcome=str(status) if status else "exception",
            ).inc()

            if self.retry_policy.exhausted(attempt):
                assert self._redis is not None
                await dead_letter(
                    self._redis,
                    event.event_id,
                    event.merchant_id,
                    f"exhausted status={status}",
                )
                SENT.labels(merchant_id=event.merchant_id, outcome="dlq").inc()
                await self._ack(msg_id)
                return

            delay = self.retry_policy.backoff(attempt)
            await asyncio.sleep(delay)

    async def _ack(self, msg_id: str) -> None:
        assert self._redis is not None
        await self._redis.xack(STREAM, GROUP, msg_id)

    @staticmethod
    def _parse(fields: dict) -> SettlementEvent:
        return SettlementEvent(
            event_id=fields["event_id"],
            merchant_id=fields["merchant_id"],
            amount_cents=int(fields["amount_cents"]),
            currency=fields["currency"],
            settled_at=datetime.fromisoformat(fields["settled_at"]),
            idempotency_key=fields["idempotency_key"],
        )

    async def run(self) -> None:
        await self.start()
        sem = asyncio.Semaphore(self.concurrency)

        async def _wrap(msg_id: str, ev: SettlementEvent) -> None:
            async with sem:
                await self._dispatch(msg_id, ev)

        try:
            while not self._stop.is_set():
                assert self._redis is not None
                entries = await self._redis.xreadgroup(
                    GROUP,
                    CONSUMER,
                    {STREAM: ">"},
                    count=self.concurrency,
                    block=1000,
                )
                if not entries:
                    continue
                tasks = []
                for _stream, msgs in entries:
                    for msg_id, fields in msgs:
                        tasks.append(asyncio.create_task(_wrap(msg_id, self._parse(fields))))
                if tasks:
                    await asyncio.gather(*tasks, return_exceptions=True)
        finally:
            await self.stop()


async def _amain() -> None:
    start_http_server(int(os.environ.get("METRICS_PORT", "9100")))
    d = WebhookDispatcher(
        redis_url=os.environ.get("REDIS_URL", "redis://localhost:6379/0"),
        db_url=os.environ.get("DB_URL", "postgresql://localhost/plaidstone"),
        concurrency=int(os.environ.get("CONCURRENCY", "32")),
    )
    loop = asyncio.get_running_loop()
    for sig_name in (signal.SIGTERM, signal.SIGINT):
        loop.add_signal_handler(sig_name, d._stop.set)
    await d.run()


if __name__ == "__main__":
    asyncio.run(_amain())
```

import time

from dispatcher.signing import sign, verify


def test_sign_and_verify_roundtrip() -> None:
    body = b'{"event_id":"evt_1"}'
    ts, header = sign("whsec_abc", body)
    assert verify("whsec_abc", body, header)
    assert not verify("whsec_wrong", body, header)


def test_reject_stale_timestamp() -> None:
    body = b'{"x":1}'
    ts, header = sign("whsec_abc", body, timestamp=int(time.time()) - 600)
    assert not verify("whsec_abc", body, header)
