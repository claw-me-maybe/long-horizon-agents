```markdown
# plaidstone-webhook-dispatcher (v1)

Async dispatcher for Plaidstone settlement webhooks (ticket PLD-2147).

## Setup
```
pip install -r requirements.txt
export REDIS_URL=redis://localhost:6379/0
export DB_URL=postgresql://plaidstone@localhost/plaidstone
python -m dispatcher.client
```

This is a v1 scaffold: unsigned POSTs, naive per-merchant semaphore, happy-path test only.
Signing, retries, rate-limiting, and metrics follow in later iterations.
```

## `dispatcher/client.py`

```python
"""V1 dispatcher: reads settlements from Redis stream, POSTs to merchant URLs.

Intentionally minimal — no signing, no retries, naive rate limit via semaphore.
"""
from __future__ import annotations

import asyncio
import json
import logging
import os
from dataclasses import asdict
from datetime import datetime
from typing import Dict

import asyncpg
import httpx
import redis.asyncio as aioredis

from events.models import SettlementEvent

log = logging.getLogger("dispatcher")

STREAM = "settlements:v2"
GROUP = "dispatcher-g1"
CONSUMER = "worker-1"


class WebhookDispatcher:
    def __init__(
        self,
        redis_url: str,
        db_url: str,
        concurrency: int = 16,
    ) -> None:
        self.redis_url = redis_url
        self.db_url = db_url
        self.concurrency = concurrency
        self._redis: aioredis.Redis | None = None
        self._db: asyncpg.Pool | None = None
        self._http: httpx.AsyncClient | None = None
        self._sems: Dict[str, asyncio.Semaphore] = {}
        self._stop = asyncio.Event()

    async def start(self) -> None:
        self._redis = aioredis.from_url(self.redis_url, decode_responses=True)
        self._db = await asyncpg.create_pool(self.db_url, min_size=1, max_size=4)
        self._http = httpx.AsyncClient(timeout=10.0)
        try:
            await self._redis.xgroup_create(STREAM, GROUP, id="0", mkstream=True)
        except Exception:
            pass  # group already exists

    async def stop(self) -> None:
        self._stop.set()
        if self._http:
            await self._http.aclose()
        if self._db:
            await self._db.close()
        if self._redis:
            await self._redis.aclose()

    async def _get_merchant(self, merchant_id: str) -> dict | None:
        assert self._db is not None
        row = await self._db.fetchrow(
            "SELECT merchant_id, url, secret, max_rps, active "
            "FROM merchant_webhooks WHERE merchant_id = $1",
            merchant_id,
        )
        return dict(row) if row else None

    def _sem_for(self, merchant_id: str, max_rps: int) -> asyncio.Semaphore:
        if merchant_id not in self._sems:
            # v1: cap concurrent sends per merchant at max_rps — crude rate limit
            self._sems[merchant_id] = asyncio.Semaphore(max(1, max_rps))
        return self._sems[merchant_id]

    async def _dispatch(self, event: SettlementEvent) -> None:
        merchant = await self._get_merchant(event.merchant_id)
        if not merchant or not merchant["active"]:
            log.info("skip inactive merchant %s", event.merchant_id)
            return
        sem = self._sem_for(event.merchant_id, merchant["max_rps"])
        async with sem:
            body = json.dumps(
                {
                    "event_id": event.event_id,
                    "merchant_id": event.merchant_id,
                    "amount_cents": event.amount_cents,
                    "currency": event.currency,
                    "settled_at": event.settled_at.isoformat(),
                    "idempotency_key": event.idempotency_key,
                }
            )
            assert self._http is not None
            resp = await self._http.post(
                merchant["url"],
                content=body,
                headers={"Content-Type": "application/json"},
            )
            log.info("dispatched %s status=%s", event.event_id, resp.status_code)
            # naive: only ack on 2xx, otherwise leave for reprocess
            if 200 <= resp.status_code < 300:
                await self._ack(event)

    async def _ack(self, event: SettlementEvent) -> None:
        assert self._redis is not None
        await self._redis.xack(STREAM, GROUP, event.event_id)

    @staticmethod
    def _parse(entry_id: str, fields: dict) -> SettlementEvent:
        return SettlementEvent(
            event_id=fields["event_id"],
            merchant_id=fields["merchant_id"],
            amount_cents=int(fields["amount_cents"]),
            currency=fields["currency"],
            settled_at=datetime.fromisoformat(fields["settled_at"]),
            idempotency_key=fields["idempotency_key"],
        )

    async def run(self) -> None:
        await self.start()
        try:
            while not self._stop.is_set():
                assert self._redis is not None
                entries = await self._redis.xreadgroup(
                    GROUP, CONSUMER, {STREAM: ">"}, count=self.concurrency, block=1000
                )
                if not entries:
                    continue
                tasks = []
                for _stream, msgs in entries:
                    for _msg_id, fields in msgs:
                        event = self._parse(_msg_id, fields)
                        tasks.append(asyncio.create_task(self._dispatch(event)))
                if tasks:
                    await asyncio.gather(*tasks, return_exceptions=True)
        finally:
            await self.stop()


async def _amain() -> None:
    logging.basicConfig(level=logging.INFO)
    d = WebhookDispatcher(
        redis_url=os.environ.get("REDIS_URL", "redis://localhost:6379/0"),
        db_url=os.environ.get("DB_URL", "postgresql://localhost/plaidstone"),
        concurrency=int(os.environ.get("CONCURRENCY", "16")),
    )
    await d.run()


if __name__ == "__main__":
    asyncio.run(_amain())
```

## `tests/test_dispatcher.py`

```python
"""Happy-path test: one event in, one POST out."""
from __future__ import annotations

import json
from datetime import datetime, timezone
from unittest.mock import AsyncMock, MagicMock

import httpx
import pytest
import respx

from dispatcher.client import WebhookDispatcher
from events.models import SettlementEvent


@pytest.mark.asyncio
@respx.mock
async def test_happy_path_dispatches_post() -> None:
    route = respx.post("https://hooks.acme-retail.com/plaidstone").mock(
        return_value=httpx.Response(200, json={"ok": True})
    )

    d = WebhookDispatcher("redis://localhost/0", "postgresql://localhost/x")
    d._http = httpx.AsyncClient()
    d._db = MagicMock()
    d._db.fetchrow = AsyncMock(
        return_value={
            "merchant_id": "merch_8f2a",
            "url": "https://hooks.acme-retail.com/plaidstone",
            "secret": "whsec_test",
            "max_rps": 5,
            "active": True,
        }
    )
    d._redis = MagicMock()
    d._redis.xack = AsyncMock(return_value=1)

    event = SettlementEvent(
        event_id="evt_1",
        merchant_id="merch_8f2a",
        amount_cents=12345,
        currency="USD",
        settled_at=datetime(2026, 4, 19, 12, 0, 0, tzinfo=timezone.utc),
        idempotency_key="idem_1",
    )
    await d._dispatch(event)
    await d._http.aclose()

    assert route.called
    body = json.loads(route.calls.last.request.content)
    assert body["event_id"] == "evt_1"
    assert body["amount_cents"] == 12345
```
