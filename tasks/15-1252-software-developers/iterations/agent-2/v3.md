# Plaidstone Webhook Dispatcher (v3)

## `README.md`

```markdown
# Plaidstone Webhook Dispatcher

Async service that delivers settlement webhooks to merchant partners.

- **Source:** Redis stream `settlements:v2`, consumer group `dispatcher-g1`.
- **Sink:** Merchant HTTPS endpoints read from Postgres `merchant_webhooks`.
- **Guarantees:** At-least-once delivery with idempotency-key dedup
  (Redis `SET NX EX` with 7-day TTL).
- **Protections:** Per-merchant token-bucket rate limit (Redis-backed),
  per-merchant circuit breaker, jittered exponential retries, dead-letter
  stream `settlements:dlq`.
- **Observability:** `structlog` JSON logs + Prometheus metrics on `/metrics`.

## Environment

| Variable                 | Default                          | Notes                                   |
|--------------------------|----------------------------------|-----------------------------------------|
| `REDIS_URL`              | `redis://redis:6379/0`           | Stream, dedup set, rate-limit, breaker  |
| `DB_URL`                 | `postgres://...`                 | `merchant_webhooks` registry            |
| `CONCURRENCY`            | `32`                             | Max in-flight deliveries                |
| `METRICS_PORT`           | `9100`                           | Prometheus scrape endpoint              |
| `CONSUMER_NAME`          | hostname                         | Unique per pod; drives Redis `XREADGROUP` |

## Signing scheme (Stripe-style)

`X-Plaidstone-Timestamp: 1713484800`
`X-Plaidstone-Signature: t=1713484800,v1=<hex-hmac-sha256>`

HMAC input is `f"{ts}.{body}"`; verification uses `hmac.compare_digest` and
rejects events older than 5 minutes.

## `dispatcher/signing.py`

```python
"""HMAC-SHA256 signing (Stripe-style `t=,v1=`) with timing-safe verification."""
from __future__ import annotations

import hmac
import time
from hashlib import sha256

SIG_HEADER = "X-Plaidstone-Signature"
TS_HEADER = "X-Plaidstone-Timestamp"
MAX_SKEW_SECONDS = 300  # 5 minutes


def _hmac_hex(secret: str, ts: int, body: bytes) -> str:
    return hmac.new(secret.encode(), f"{ts}.".encode() + body, sha256).hexdigest()


def sign(secret: str, body: bytes, timestamp: int | None = None) -> tuple[str, str]:
    """Return (timestamp_header_value, signature_header_value)."""
    ts = timestamp if timestamp is not None else int(time.time())
    return str(ts), f"t={ts},v1={_hmac_hex(secret, ts, body)}"


def verify(
    secret: str,
    body: bytes,
    timestamp_header: str,
    signature_header: str,
    *,
    now: int | None = None,
    max_skew: int = MAX_SKEW_SECONDS,
) -> bool:
    """Timing-safe verify; rejects if skew > max_skew or signature mismatches."""
    try:
        ts = int(timestamp_header)
    except (TypeError, ValueError):
        return False
    current = now if now is not None else int(time.time())
    if abs(current - ts) > max_skew:
        return False
    parts = dict(p.split("=", 1) for p in signature_header.split(",") if "=" in p)
    sent = parts.get("v1", "")
    expected = _hmac_hex(secret, ts, body)
    # Length-check before compare_digest (which also tolerates length mismatch).
    return hmac.compare_digest(sent.encode(), expected.encode())
```

## `dispatcher/retry.py`

```python
"""Retry policy: full-jitter exponential backoff + dead-letter push."""
from __future__ import annotations

import random
from dataclasses import dataclass

DLQ_STREAM = "settlements:dlq"


@dataclass(frozen=True)
class RetryPolicy:
    base_seconds: float = 2.0
    max_attempts: int = 5
    cap_seconds: float = 60.0

    def backoff(self, attempt: int, rng: random.Random | None = None) -> float:
        """Full-jitter backoff: uniform(0, min(cap, base * 2^(attempt-1)))."""
        r = rng or random
        ceiling = min(self.cap_seconds, self.base_seconds * (2 ** (attempt - 1)))
        return r.uniform(0.0, ceiling)

    def should_retry(
        self,
        attempt: int,
        status: int | None,
        exc: Exception | None,
    ) -> bool:
        if attempt >= self.max_attempts:
            return False
        if exc is not None:
            return True  # transport error
        if status is None:
            return True
        # Retry on 5xx, 429; drop on 4xx (client error is permanent).
        return 500 <= status < 600 or status == 429


async def push_to_dlq(redis_client, body: bytes, *, reason: str, merchant_id: str) -> None:
    await redis_client.xadd(
        DLQ_STREAM,
        {"body": body, "reason": reason, "merchant_id": merchant_id},
        maxlen=100_000,
        approximate=True,
    )
```

## `dispatcher/ratelimit.py`

```python
"""Distributed token bucket backed by Redis INCR+EXPIRE.

Design
------
A token bucket per (merchant, epoch-second). `acquire()` atomically
increments the counter; if the post-INCR value exceeds `max_rps`, the
caller sleeps until the next 1-second boundary and retries. Keys
auto-expire after 2s so stale state never leaks. State lives in Redis
rather than process memory, so multiple dispatcher pods share a single
rate budget and restarts do not reset the bucket mid-window.

Correctness under concurrency: INCR is atomic, so N concurrent tasks
will see N distinct post-increment values, and at most `max_rps` of them
will be <= `max_rps`. All others wait for the next bucket.
"""
from __future__ import annotations

import asyncio
import time


class RedisTokenBucket:
    def __init__(self, redis_client, prefix: str = "rl") -> None:
        self.r = redis_client
        self.prefix = prefix

    async def acquire(self, merchant_id: str, max_rps: int) -> None:
        assert max_rps > 0
        while True:
            now = time.time()
            bucket = int(now)
            key = f"{self.prefix}:{merchant_id}:{bucket}"
            count = await self.r.incr(key)
            if count == 1:
                # Set TTL on the first increment; EXPIRE is idempotent-cheap
                # enough that we could do it unconditionally, but this saves
                # one round trip on the steady-state path.
                await self.r.expire(key, 2)
            if count <= max_rps:
                return
            sleep = max(0.0, (bucket + 1) - now)
            await asyncio.sleep(sleep)
```

## `dispatcher/breaker.py`

```python
"""Per-merchant circuit breaker backed by Redis.

States
------
- CLOSED:    normal; failures tick a counter.
- OPEN:      fast-fail; counter reset after cooldown.
- HALF_OPEN: allow one probe; success -> CLOSED, failure -> OPEN.

Storage
-------
- `cb:fail:{merchant}`  -> int, consecutive 5xx count (closed state)
- `cb:open:{merchant}`  -> str "1", TTL = cooldown (presence == OPEN)
- `cb:probe:{merchant}` -> str "1", TTL = probe window (claim the probe slot)
"""
from __future__ import annotations


class CircuitBreaker:
    def __init__(
        self,
        redis_client,
        *,
        threshold: int = 5,
        cooldown_seconds: int = 30,
        probe_window_seconds: int = 5,
    ) -> None:
        self.r = redis_client
        self.threshold = threshold
        self.cooldown = cooldown_seconds
        self.probe_window = probe_window_seconds

    def _fail_key(self, mid: str) -> str:
        return f"cb:fail:{mid}"

    def _open_key(self, mid: str) -> str:
        return f"cb:open:{mid}"

    def _probe_key(self, mid: str) -> str:
        return f"cb:probe:{mid}"

    async def allow(self, merchant_id: str) -> bool:
        """Return True if a call is permitted, False if the circuit is open."""
        is_open = await self.r.get(self._open_key(merchant_id))
        if not is_open:
            return True
        # OPEN: try to claim the single probe slot.
        claimed = await self.r.set(
            self._probe_key(merchant_id), "1", nx=True, ex=self.probe_window
        )
        return bool(claimed)

    async def record_success(self, merchant_id: str) -> None:
        await self.r.delete(self._fail_key(merchant_id),
                            self._open_key(merchant_id),
                            self._probe_key(merchant_id))

    async def record_failure(self, merchant_id: str) -> None:
        """Increment failure counter; trip to OPEN at threshold."""
        # If we were half-open, a failure immediately re-opens the circuit.
        count = await self.r.incr(self._fail_key(merchant_id))
        if count >= self.threshold:
            await self.r.set(self._open_key(merchant_id), "1", ex=self.cooldown)
            await self.r.delete(self._probe_key(merchant_id))
```

## `dispatcher/client.py`

```python
"""WebhookDispatcher: async loop, rate limit, circuit breaker, retries, metrics."""
from __future__ import annotations

import asyncio
import json
import os
import signal
import socket
import time
from typing import Any

import asyncpg
import httpx
import redis.asyncio as redis
import structlog
from prometheus_client import start_http_server

from dispatcher.breaker import CircuitBreaker
from dispatcher.metrics import (
    webhook_latency_seconds,
    webhook_retries_total,
    webhooks_sent_total,
)
from dispatcher.models import MerchantConfig, SettlementEvent
from dispatcher.ratelimit import RedisTokenBucket
from dispatcher.retry import RetryPolicy, push_to_dlq
from dispatcher.signing import SIG_HEADER, TS_HEADER, sign

log = structlog.get_logger("dispatcher")

DEDUP_KEY = "idem:{}"
DEDUP_TTL_SECONDS = 7 * 24 * 3600


class WebhookDispatcher:
    def __init__(
        self,
        redis_client: redis.Redis,
        pg_pool: asyncpg.Pool | Any,
        http_client: httpx.AsyncClient,
        *,
        concurrency: int = 32,
        stream: str = "settlements:v2",
        group: str = "dispatcher-g1",
        consumer: str | None = None,
        retry_policy: RetryPolicy | None = None,
        breaker: CircuitBreaker | None = None,
    ) -> None:
        self.r = redis_client
        self.pg = pg_pool
        self.http = http_client
        self.stream = stream
        self.group = group
        self.consumer = consumer or socket.gethostname()
        self.sem = asyncio.Semaphore(concurrency)
        self.bucket = RedisTokenBucket(redis_client)
        self.retry = retry_policy or RetryPolicy()
        self.breaker = breaker or CircuitBreaker(redis_client)
        self._stop = asyncio.Event()

    # --- lifecycle ---------------------------------------------------------

    async def start(self) -> None:
        try:
            await self.r.xgroup_create(self.stream, self.group, id="0", mkstream=True)
        except redis.ResponseError as e:
            if "BUSYGROUP" not in str(e):
                raise

    def request_stop(self) -> None:
        self._stop.set()

    # --- helpers -----------------------------------------------------------

    async def _load_merchant(self, merchant_id: str) -> MerchantConfig | None:
        row = await self.pg.fetchrow(
            "SELECT merchant_id, url, secret, max_rps, active "
            "FROM merchant_webhooks WHERE merchant_id = $1",
            merchant_id,
        )
        return MerchantConfig(**dict(row)) if row else None

    async def _seen_before(self, idempotency_key: str) -> bool:
        """Atomic SET NX EX; returns True if key already existed (duplicate)."""
        created = await self.r.set(
            DEDUP_KEY.format(idempotency_key),
            "1",
            nx=True,
            ex=DEDUP_TTL_SECONDS,
        )
        return not bool(created)

    async def _post_once(self, merchant: MerchantConfig, body: bytes) -> httpx.Response:
        ts_h, sig_h = sign(merchant.secret, body)
        t0 = time.perf_counter()
        try:
            return await self.http.post(
                merchant.url,
                content=body,
                headers={
                    "Content-Type": "application/json",
                    TS_HEADER: ts_h,
                    SIG_HEADER: sig_h,
                },
            )
        finally:
            webhook_latency_seconds.labels(
                merchant_id=merchant.merchant_id
            ).observe(time.perf_counter() - t0)

    # --- delivery pipeline -------------------------------------------------

    async def _deliver_with_retries(
        self, merchant: MerchantConfig, body: bytes
    ) -> bool:
        attempt = 0
        last_status: int | None = None
        last_exc: Exception | None = None
        while True:
            attempt += 1
            if not await self.breaker.allow(merchant.merchant_id):
                webhooks_sent_total.labels(
                    merchant_id=merchant.merchant_id, outcome="circuit_open"
                ).inc()
                log.warning("breaker.open", merchant_id=merchant.merchant_id)
                return False

            await self.bucket.acquire(merchant.merchant_id, merchant.max_rps)

            try:
                resp = await self._post_once(merchant, body)
            except httpx.HTTPError as exc:
                last_exc, last_status = exc, None
            else:
                last_exc = None
                last_status = resp.status_code
                if 200 <= resp.status_code < 300:
                    await self.breaker.record_success(merchant.merchant_id)
                    if attempt > 1:
                        webhook_retries_total.labels(
                            merchant_id=merchant.merchant_id, outcome="success"
                        ).inc()
                    return True

            # Failure path: bookkeeping.
            if last_exc is not None or (last_status and last_status >= 500):
                await self.breaker.record_failure(merchant.merchant_id)

            if not self.retry.should_retry(attempt, last_status, last_exc):
                webhook_retries_total.labels(
                    merchant_id=merchant.merchant_id, outcome="give_up"
                ).inc()
                return False

            webhook_retries_total.labels(
                merchant_id=merchant.merchant_id, outcome="failure"
            ).inc()
            await asyncio.sleep(self.retry.backoff(attempt))

    async def _process(self, entry_id: str, event: SettlementEvent) -> None:
        async with self.sem:
            try:
                if await self._seen_before(event.idempotency_key):
                    webhooks_sent_total.labels(
                        merchant_id=event.merchant_id, outcome="dedup"
                    ).inc()
                    log.info(
                        "dedup.drop",
                        event_id=event.event_id,
                        idempotency_key=event.idempotency_key,
                    )
                    return

                merchant = await self._load_merchant(event.merchant_id)
                if merchant is None or not merchant.active:
                    log.warning("merchant.missing", merchant_id=event.merchant_id)
                    webhooks_sent_total.labels(
                        merchant_id=event.merchant_id, outcome="failure"
                    ).inc()
                    return

                body = _event_to_json(event)
                ok = await self._deliver_with_retries(merchant, body)
                if ok:
                    webhooks_sent_total.labels(
                        merchant_id=merchant.merchant_id, outcome="success"
                    ).inc()
                    log.info(
                        "delivery.ok",
                        event_id=event.event_id,
                        merchant_id=merchant.merchant_id,
                    )
                else:
                    await push_to_dlq(
                        self.r, body,
                        reason="retry_exhausted",
                        merchant_id=merchant.merchant_id,
                    )
                    webhooks_sent_total.labels(
                        merchant_id=merchant.merchant_id, outcome="dlq"
                    ).inc()
                    log.error(
                        "delivery.dlq",
                        event_id=event.event_id,
                        merchant_id=merchant.merchant_id,
                    )
            except Exception:  # pragma: no cover - defensive
                log.exception("delivery.error", event_id=event.event_id)
            finally:
                # ACK regardless: duplicate / dead-lettered events must not
                # re-circulate through the main stream.
                await self.r.xack(self.stream, self.group, entry_id)

    # --- main loop ---------------------------------------------------------

    async def run(self) -> None:
        await self.start()
        log.info("dispatcher.started",
                 stream=self.stream, group=self.group, consumer=self.consumer)
        while not self._stop.is_set():
            try:
                resp = await self.r.xreadgroup(
                    self.group, self.consumer,
                    streams={self.stream: ">"},
                    count=64,
                    block=1000,
                )
            except redis.ResponseError as e:
                log.error("xreadgroup.error", err=str(e))
                await asyncio.sleep(1)
                continue
            if not resp:
                continue
            tasks = [
                self._process(entry_id, _parse_event(data))
                for _stream, entries in resp
                for entry_id, data in entries
            ]
            await asyncio.gather(*tasks, return_exceptions=True)
        log.info("dispatcher.stopped")


# --- serialization -----------------------------------------------------------


def _event_to_json(event: SettlementEvent) -> bytes:
    return json.dumps(
        {
            "event_id": event.event_id,
            "merchant_id": event.merchant_id,
            "amount_cents": event.amount_cents,
            "currency": event.currency,
            "settled_at": event.settled_at.isoformat(),
            "idempotency_key": event.idempotency_key,
        },
        sort_keys=True,
        separators=(",", ":"),
    ).encode()


def _parse_event(data: dict[str, Any]) -> SettlementEvent:
    from datetime import datetime

    return SettlementEvent(
        event_id=data["event_id"],
        merchant_id=data["merchant_id"],
        amount_cents=int(data["amount_cents"]),
        currency=data["currency"],
        settled_at=datetime.fromisoformat(data["settled_at"]),
        idempotency_key=data["idempotency_key"],
    )


# --- entrypoint --------------------------------------------------------------


async def _main() -> None:
    structlog.configure(
        processors=[
            structlog.processors.add_log_level,
            structlog.processors.TimeStamper(fmt="iso"),
            structlog.processors.JSONRenderer(),
        ]
    )
    start_http_server(int(os.environ.get("METRICS_PORT", "9100")))

    r = redis.from_url(os.environ["REDIS_URL"], decode_responses=True)
    pg = await asyncpg.create_pool(os.environ["DB_URL"], min_size=1, max_size=4)
    http = httpx.AsyncClient(timeout=10.0)

    dispatcher = WebhookDispatcher(
        r, pg, http,
        concurrency=int(os.environ.get("CONCURRENCY", "32")),
        consumer=os.environ.get("CONSUMER_NAME") or socket.gethostname(),
    )

    loop = asyncio.get_running_loop()
    for sig in (signal.SIGTERM, signal.SIGINT):
        loop.add_signal_handler(sig, dispatcher.request_stop)

    try:
        await dispatcher.run()
    finally:
        await http.aclose()
        await pg.close()
        await r.close()


if __name__ == "__main__":
    asyncio.run(_main())
```

## `tests/test_signing.py`

```python
"""Signing correctness + timing-attack resistance + skew rejection."""
from __future__ import annotations

import time

from dispatcher.signing import MAX_SKEW_SECONDS, sign, verify


def test_sign_then_verify_roundtrip() -> None:
    body = b'{"event_id": "evt_1"}'
    ts_h, sig_h = sign("whsec_test", body)
    assert verify("whsec_test", body, ts_h, sig_h)


def test_verify_rejects_stale_timestamp() -> None:
    body = b'{}'
    ts_h, sig_h = sign("s", body, timestamp=int(time.time()) - MAX_SKEW_SECONDS - 1)
    assert not verify("s", body, ts_h, sig_h)


def test_verify_rejects_tampered_body() -> None:
    ts_h, sig_h = sign("s", b"original")
    assert not verify("s", b"tampered", ts_h, sig_h)


def test_verify_rejects_wrong_secret() -> None:
