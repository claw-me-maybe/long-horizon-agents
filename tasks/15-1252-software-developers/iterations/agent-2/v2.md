# Plaidstone Webhook Dispatcher (v2)

## `README.md`

```markdown
# Plaidstone Webhook Dispatcher

Async dispatcher for settlement webhooks. Consumes from the Redis stream
`settlements:v2` (consumer group `dispatcher-g1`), looks up merchant config
in Postgres, signs the payload, and POSTs with exponential-backoff retries.

## Environment

| Var            | Default                        | Purpose                       |
|----------------|--------------------------------|-------------------------------|
| `REDIS_URL`    | `redis://redis:6379/0`         | Stream + rate-limit state     |
| `DB_URL`       | `postgres://...`               | `merchant_webhooks` registry  |
| `CONCURRENCY`  | `32`                           | Max in-flight deliveries      |

## Install & run

```bash
pip install -r requirements.txt
python -m dispatcher.client
```

docker compose up

## `dispatcher/signing.py`

```python
"""HMAC-SHA256 signing helpers (Stripe-style `t=,v1=` scheme)."""
from __future__ import annotations

import hmac
import time
from hashlib import sha256

SIG_HEADER = "X-Plaidstone-Signature"
TS_HEADER = "X-Plaidstone-Timestamp"
MAX_SKEW_SECONDS = 300  # 5 min


def sign(secret: str, body: bytes, timestamp: int | None = None) -> tuple[str, str]:
    """Return (timestamp_header, signature_header) for `body`."""
    ts = timestamp if timestamp is not None else int(time.time())
    mac = hmac.new(secret.encode(), f"{ts}.".encode() + body, sha256).hexdigest()
    return str(ts), f"t={ts},v1={mac}"


def verify(secret: str, body: bytes, timestamp_header: str, signature_header: str,
           now: int | None = None, max_skew: int = MAX_SKEW_SECONDS) -> bool:
    """Timing-safe verify; rejects if skew > max_skew."""
    try:
        ts = int(timestamp_header)
    except ValueError:
        return False
    current = now if now is not None else int(time.time())
    if abs(current - ts) > max_skew:
        return False
    parts = dict(p.split("=", 1) for p in signature_header.split(",") if "=" in p)
    sent = parts.get("v1", "")
    expected = hmac.new(secret.encode(), f"{ts}.".encode() + body, sha256).hexdigest()
    return hmac.compare_digest(sent, expected)
```

## `dispatcher/retry.py`

```python
"""Retry policy: jittered exponential backoff + dead-letter push."""
from __future__ import annotations

import random
from dataclasses import dataclass

DLQ_STREAM = "settlements:dlq"


@dataclass(frozen=True)
class RetryPolicy:
    base_seconds: float = 2.0
    max_attempts: int = 5
    cap_seconds: float = 60.0

    def backoff(self, attempt: int) -> float:
        """Return sleep seconds before `attempt` (1-indexed)."""
        raw = min(self.cap_seconds, self.base_seconds * (2 ** (attempt - 1)))
        return random.uniform(0.0, raw)  # full jitter

    def should_retry(self, attempt: int, status: int | None, exc: Exception | None) -> bool:
        if attempt >= self.max_attempts:
            return False
        if exc is not None:
            return True
        if status is None:
            return True
        return 500 <= status < 600 or status == 429


async def push_to_dlq(redis_client, event_json: bytes, reason: str) -> None:
    await redis_client.xadd(DLQ_STREAM, {"body": event_json, "reason": reason})
```

## `dispatcher/ratelimit.py`

```python
"""Distributed token bucket backed by Redis INCR + EXPIRE.

One bucket = one 1-second window keyed as `rl:{merchant_id}:{epoch_second}`.
`acquire()` INCRs the counter; if the post-INCR value exceeds `max_rps`,
the caller sleeps until the next second. Bucket keys auto-expire after 2s
so stale state never leaks. Survives dispatcher restarts because state
lives in Redis, not process memory.
"""
from __future__ import annotations

import asyncio
import time


class RedisTokenBucket:
    def __init__(self, redis_client, prefix: str = "rl") -> None:
        self.r = redis_client
        self.prefix = prefix

    async def acquire(self, merchant_id: str, max_rps: int) -> None:
        while True:
            now = time.time()
            bucket = int(now)
            key = f"{self.prefix}:{merchant_id}:{bucket}"
            count = await self.r.incr(key)
            if count == 1:
                await self.r.expire(key, 2)
            if count <= max_rps:
                return
            # Wait until next 1s window, then retry.
            await asyncio.sleep(max(0.0, (bucket + 1) - now))
```

## `dispatcher/client.py`

```python
"""WebhookDispatcher: async loop, Redis token-bucket, signed POSTs, retries."""
from __future__ import annotations

import asyncio
import json
import os
import signal
from dataclasses import dataclass
from datetime import datetime
from typing import Any

import asyncpg
import httpx
import redis.asyncio as redis
import structlog

from dispatcher.ratelimit import RedisTokenBucket
from dispatcher.retry import RetryPolicy, push_to_dlq
from dispatcher.signing import SIG_HEADER, TS_HEADER, sign

log = structlog.get_logger("dispatcher")

DEDUP_KEY = "idem:{}"
DEDUP_TTL_SECONDS = 7 * 24 * 3600


@dataclass
class MerchantConfig:
    merchant_id: str
    url: str
    secret: str
    max_rps: int
    active: bool


@dataclass
class SettlementEvent:
    event_id: str
    merchant_id: str
    amount_cents: int
    currency: str
    settled_at: datetime
    idempotency_key: str


class WebhookDispatcher:
    def __init__(
        self,
        redis_client: redis.Redis,
        pg_pool: asyncpg.Pool,
        http_client: httpx.AsyncClient,
        *,
        concurrency: int = 32,
        stream: str = "settlements:v2",
        group: str = "dispatcher-g1",
        consumer: str = "dispatcher-1",
        retry_policy: RetryPolicy | None = None,
    ) -> None:
        self.r = redis_client
        self.pg = pg_pool
        self.http = http_client
        self.stream = stream
        self.group = group
        self.consumer = consumer
        self.sem = asyncio.Semaphore(concurrency)
        self.bucket = RedisTokenBucket(redis_client)
        self.retry = retry_policy or RetryPolicy()
        self._stop = asyncio.Event()

    async def start(self) -> None:
        try:
            await self.r.xgroup_create(self.stream, self.group, id="0", mkstream=True)
        except redis.ResponseError as e:
            if "BUSYGROUP" not in str(e):
                raise

    def request_stop(self) -> None:
        self._stop.set()

    async def _load_merchant(self, merchant_id: str) -> MerchantConfig | None:
        row = await self.pg.fetchrow(
            "SELECT merchant_id, url, secret, max_rps, active "
            "FROM merchant_webhooks WHERE merchant_id = $1",
            merchant_id,
        )
        return MerchantConfig(**dict(row)) if row else None

    async def _seen_before(self, idempotency_key: str) -> bool:
        key = DEDUP_KEY.format(idempotency_key)
        created = await self.r.set(key, "1", nx=True, ex=DEDUP_TTL_SECONDS)
        return not bool(created)

    async def _post_once(self, merchant: MerchantConfig, body: bytes) -> httpx.Response:
        ts_h, sig_h = sign(merchant.secret, body)
        return await self.http.post(
            merchant.url,
            content=body,
            headers={
                "Content-Type": "application/json",
                TS_HEADER: ts_h,
                SIG_HEADER: sig_h,
            },
        )

    async def _deliver_with_retries(self, merchant: MerchantConfig, body: bytes) -> bool:
        attempt = 0
        last_status: int | None = None
        last_exc: Exception | None = None
        while True:
            attempt += 1
            await self.bucket.acquire(merchant.merchant_id, merchant.max_rps)
            try:
                resp = await self._post_once(merchant, body)
            except httpx.HTTPError as exc:
                last_exc, last_status = exc, None
            else:
                last_exc = None
                last_status = resp.status_code
                if 200 <= resp.status_code < 300:
                    return True
            if not self.retry.should_retry(attempt, last_status, last_exc):
                return False
            await asyncio.sleep(self.retry.backoff(attempt))

    async def _process(self, entry_id: str, event: SettlementEvent) -> None:
        async with self.sem:
            try:
                if await self._seen_before(event.idempotency_key):
                    log.info("dedup.drop", event_id=event.event_id,
                             idempotency_key=event.idempotency_key)
                    return
                merchant = await self._load_merchant(event.merchant_id)
                if merchant is None or not merchant.active:
                    log.warning("merchant.missing", merchant_id=event.merchant_id)
                    return
                body = _event_to_json(event)
                ok = await self._deliver_with_retries(merchant, body)
                if not ok:
                    await push_to_dlq(self.r, body, "retry_exhausted")
                    log.error("delivery.dlq", event_id=event.event_id)
                else:
                    log.info("delivery.ok", event_id=event.event_id,
                             merchant_id=event.merchant_id)
            finally:
                await self.r.xack(self.stream, self.group, entry_id)

    async def run(self) -> None:
        await self.start()
        while not self._stop.is_set():
            try:
                resp = await self.r.xreadgroup(
                    self.group, self.consumer,
                    streams={self.stream: ">"},
                    count=64,
                    block=1000,
                )
            except redis.ResponseError as e:
                log.error("xreadgroup.error", err=str(e))
                await asyncio.sleep(1)
                continue
            if not resp:
                continue
            tasks = [
                self._process(entry_id, _parse_event(data))
                for _stream, entries in resp
                for entry_id, data in entries
            ]
            await asyncio.gather(*tasks, return_exceptions=True)


def _event_to_json(event: SettlementEvent) -> bytes:
    return json.dumps({
        "event_id": event.event_id,
        "merchant_id": event.merchant_id,
        "amount_cents": event.amount_cents,
        "currency": event.currency,
        "settled_at": event.settled_at.isoformat(),
        "idempotency_key": event.idempotency_key,
    }, sort_keys=True, separators=(",", ":")).encode()


def _parse_event(data: dict[str, Any]) -> SettlementEvent:
    return SettlementEvent(
        event_id=data["event_id"],
        merchant_id=data["merchant_id"],
        amount_cents=int(data["amount_cents"]),
        currency=data["currency"],
        settled_at=datetime.fromisoformat(data["settled_at"]),
        idempotency_key=data["idempotency_key"],
    )


async def _main() -> None:
    structlog.configure(processors=[structlog.processors.JSONRenderer()])
    r = redis.from_url(os.environ["REDIS_URL"], decode_responses=True)
    pg = await asyncpg.create_pool(os.environ["DB_URL"], min_size=1, max_size=4)
    http = httpx.AsyncClient(timeout=10.0)
    dispatcher = WebhookDispatcher(
        r, pg, http,
        concurrency=int(os.environ.get("CONCURRENCY", "32")),
    )

    loop = asyncio.get_running_loop()
    for sig in (signal.SIGTERM, signal.SIGINT):
        loop.add_signal_handler(sig, dispatcher.request_stop)

    try:
        await dispatcher.run()
    finally:
        await http.aclose()
        await pg.close()
        await r.close()


if __name__ == "__main__":
    asyncio.run(_main())
```

## `tests/test_dispatcher.py`

```python
"""Dispatcher + signing + rate-limit tests using fakeredis and respx."""
from __future__ import annotations

import asyncio
import time
from datetime import datetime, timezone

import httpx
import pytest
import respx

from dispatcher.client import (
    MerchantConfig,
    SettlementEvent,
    WebhookDispatcher,
    _event_to_json,
)
from dispatcher.signing import SIG_HEADER, TS_HEADER, sign, verify


def _make_event(event_id: str = "evt_1", idem: str = "idem_1") -> SettlementEvent:
    return SettlementEvent(
        event_id=event_id,
        merchant_id="merch_8f2a",
        amount_cents=12_345,
        currency="USD",
        settled_at=datetime(2026, 4, 19, tzinfo=timezone.utc),
        idempotency_key=idem,
    )


def _merchant(max_rps: int = 10) -> MerchantConfig:
    return MerchantConfig(
        merchant_id="merch_8f2a",
        url="https://hooks.acme-retail.com/plaidstone",
        secret="whsec_test",
        max_rps=max_rps,
        active=True,
    )


def test_sign_then_verify_roundtrip() -> None:
    body = b'{"k":"v"}'
    ts_h, sig_h = sign("whsec_test", body)
    assert verify("whsec_test", body, ts_h, sig_h)


def test_verify_rejects_stale_timestamp() -> None:
    body = b'{}'
    ts_h, sig_h = sign("s", body, timestamp=int(time.time()) - 3600)
    assert not verify("s", body, ts_h, sig_h)


@pytest.mark.asyncio
async def test_duplicate_events_are_dropped(redis_client, pg_pool, http_client) -> None:
    dispatcher = WebhookDispatcher(redis_client, pg_pool, http_client)
    event = _make_event()
    body = _event_to_json(event)

    with respx.mock() as mock:
        route = mock.post("https://hooks.acme-retail.com/plaidstone").mock(
            return_value=httpx.Response(200)
        )
        ok1 = await dispatcher._deliver_with_retries(_merchant(), body)
        assert ok1 is True
        # First delivery "succeeded"; now mark idem key and retry.
        assert await dispatcher._seen_before(event.idempotency_key) is False
        assert await dispatcher._seen_before(event.idempotency_key) is True

    assert route.call_count == 1  # second attempt skipped at _process level


@pytest.mark.asyncio
async def test_rate_limit_holds_under_concurrency(redis_client, pg_pool, http_client) -> None:
    dispatcher = WebhookDispatcher(redis_client, pg_pool, http_client)
    merchant = _merchant(max_rps=10)

    with respx.mock() as mock:
        mock.post(merchant.url).mock(return_value=httpx.Response(200))
        start = time.monotonic()
        bodies = [_event_to_json(_make_event(f"evt_{i}", f"idem_{i}")) for i in range(50)]
        await asyncio.gather(*[
            dispatcher._deliver_with_retries(merchant, b) for b in bodies
        ])
        elapsed = time.monotonic() - start

    # 50 reqs / 10 rps = at least 4s wall clock (first 10 in bucket 0,
    # next 10 in bucket 1, etc.).
    assert elapsed >= 4.0


@pytest.mark.asyncio
async def test_retry_then_dlq_on_persistent_5xx(redis_client, pg_pool, http_client) -> None:
    dispatcher = WebhookDispatcher(redis_client, pg_pool, http_client)
    dispatcher.retry = dispatcher.retry.__class__(base_seconds=0.0, cap_seconds=0.0, max_attempts=3)
    merchant = _merchant()
    body = b'{"test": true}'

    with respx.mock() as mock:
        route = mock.post(merchant.url).mock(return_value=httpx.Response(503))
        ok = await dispatcher._deliver_with_retries(merchant, body)

    assert ok is False
    assert route.call_count == 3
