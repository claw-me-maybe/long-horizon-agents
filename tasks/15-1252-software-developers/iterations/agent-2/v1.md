# Webhook Dispatcher

Minimal async dispatcher for Plaidstone settlement webhooks (PLD-2147).

## Run

```bash
pip install -r requirements.txt
export REDIS_URL=redis://localhost:6379/0
export DB_URL=postgresql://plaidstone:plaidstone@localhost:5432/plaidstone
python -m dispatcher.client
```

Consumes from Redis stream `settlements:v2` (group `dispatcher-g1`), looks up
merchant URLs in Postgres `merchant_webhooks`, POSTs JSON. Rate limiting is
naive per-merchant semaphore; signing/retries come later.
```

## `dispatcher/client.py`

```python
"""Basic async webhook dispatcher (v1).

No signing, naive per-merchant concurrency limit via asyncio.Semaphore,
single POST attempt, no retries.
"""
from __future__ import annotations

import asyncio
import json
import logging
import os
from dataclasses import dataclass
from datetime import datetime
from typing import Any

import asyncpg
import httpx
import redis.asyncio as redis

log = logging.getLogger("dispatcher")


@dataclass
class MerchantConfig:
    merchant_id: str
    url: str
    secret: str
    max_rps: int
    active: bool


@dataclass
class SettlementEvent:
    event_id: str
    merchant_id: str
    amount_cents: int
    currency: str
    settled_at: datetime
    idempotency_key: str


class WebhookDispatcher:
    def __init__(
        self,
        redis_url: str,
        db_url: str,
        concurrency: int = 16,
        stream: str = "settlements:v2",
        group: str = "dispatcher-g1",
        consumer: str = "dispatcher-1",
    ) -> None:
        self.redis_url = redis_url
        self.db_url = db_url
        self.stream = stream
        self.group = group
        self.consumer = consumer
        self.concurrency = concurrency
        self._sems: dict[str, asyncio.Semaphore] = {}
        self._redis: redis.Redis | None = None
        self._pg: asyncpg.Pool | None = None
        self._http: httpx.AsyncClient | None = None
        self._stopping = asyncio.Event()

    async def start(self) -> None:
        self._redis = redis.from_url(self.redis_url, decode_responses=True)
        self._pg = await asyncpg.create_pool(self.db_url, min_size=1, max_size=4)
        self._http = httpx.AsyncClient(timeout=10.0)
        try:
            await self._redis.xgroup_create(self.stream, self.group, id="0", mkstream=True)
        except redis.ResponseError:
            pass  # group already exists

    async def stop(self) -> None:
        self._stopping.set()
        if self._http:
            await self._http.aclose()
        if self._pg:
            await self._pg.close()
        if self._redis:
            await self._redis.close()

    def _sem(self, merchant_id: str, max_rps: int) -> asyncio.Semaphore:
        if merchant_id not in self._sems:
            self._sems[merchant_id] = asyncio.Semaphore(max(1, max_rps))
        return self._sems[merchant_id]

    async def _load_merchant(self, merchant_id: str) -> MerchantConfig | None:
        assert self._pg is not None
        row = await self._pg.fetchrow(
            "SELECT merchant_id, url, secret, max_rps, active "
            "FROM merchant_webhooks WHERE merchant_id = $1",
            merchant_id,
        )
        if not row:
            return None
        return MerchantConfig(**dict(row))

    async def _deliver(self, event: SettlementEvent) -> None:
        assert self._http is not None
        merchant = await self._load_merchant(event.merchant_id)
        if merchant is None or not merchant.active:
            log.warning("no-active-merchant %s", event.merchant_id)
            return
        sem = self._sem(merchant.merchant_id, merchant.max_rps)
        async with sem:
            body = _event_to_json(event)
            try:
                resp = await self._http.post(merchant.url, content=body,
                                             headers={"Content-Type": "application/json"})
                log.info("posted %s -> %s status=%s", event.event_id, merchant.url, resp.status_code)
            except httpx.HTTPError as e:
                log.error("http-error %s: %s", event.event_id, e)

    async def _read_loop(self) -> None:
        assert self._redis is not None
        while not self._stopping.is_set():
            try:
                resp = await self._redis.xreadgroup(
                    self.group, self.consumer,
                    streams={self.stream: ">"},
                    count=self.concurrency,
                    block=2000,
                )
            except redis.ResponseError as e:
                log.error("xreadgroup: %s", e)
                await asyncio.sleep(1)
                continue
            if not resp:
                continue
            tasks = []
            for _stream, entries in resp:
                for entry_id, data in entries:
                    event = _parse_event(data)
                    tasks.append(self._process(entry_id, event))
            if tasks:
                await asyncio.gather(*tasks, return_exceptions=True)

    async def _process(self, entry_id: str, event: SettlementEvent) -> None:
        assert self._redis is not None
        try:
            await self._deliver(event)
        finally:
            await self._redis.xack(self.stream, self.group, entry_id)

    async def run(self) -> None:
        await self.start()
        try:
            await self._read_loop()
        finally:
            await self.stop()


def _event_to_json(event: SettlementEvent) -> bytes:
    return json.dumps({
        "event_id": event.event_id,
        "merchant_id": event.merchant_id,
        "amount_cents": event.amount_cents,
        "currency": event.currency,
        "settled_at": event.settled_at.isoformat(),
        "idempotency_key": event.idempotency_key,
    }).encode()


def _parse_event(data: dict[str, Any]) -> SettlementEvent:
    return SettlementEvent(
        event_id=data["event_id"],
        merchant_id=data["merchant_id"],
        amount_cents=int(data["amount_cents"]),
        currency=data["currency"],
        settled_at=datetime.fromisoformat(data["settled_at"]),
        idempotency_key=data["idempotency_key"],
    )


async def _main() -> None:
    logging.basicConfig(level=logging.INFO)
    dispatcher = WebhookDispatcher(
        redis_url=os.environ["REDIS_URL"],
        db_url=os.environ["DB_URL"],
        concurrency=int(os.environ.get("CONCURRENCY", "16")),
    )
    await dispatcher.run()


if __name__ == "__main__":
    asyncio.run(_main())
```

## `tests/test_dispatcher.py`

```python
"""Happy-path smoke test for v1 dispatcher."""
from __future__ import annotations

from datetime import datetime, timezone
from unittest.mock import AsyncMock, MagicMock

import httpx
import pytest
import respx

from dispatcher.client import (
    MerchantConfig,
    SettlementEvent,
    WebhookDispatcher,
    _event_to_json,
)


@pytest.mark.asyncio
async def test_deliver_posts_to_merchant_url() -> None:
    dispatcher = WebhookDispatcher(
        redis_url="redis://localhost/0",
        db_url="postgres://x",
    )
    dispatcher._http = httpx.AsyncClient()
    dispatcher._load_merchant = AsyncMock(  # type: ignore[assignment]
        return_value=MerchantConfig(
            merchant_id="merch_8f2a",
            url="https://hooks.acme-retail.com/plaidstone",
            secret="whsec_test",
            max_rps=5,
            active=True,
        )
    )
    event = SettlementEvent(
        event_id="evt_1",
        merchant_id="merch_8f2a",
        amount_cents=12_345,
        currency="USD",
        settled_at=datetime.now(timezone.utc),
        idempotency_key="idem_1",
    )

    with respx.mock(base_url="https://hooks.acme-retail.com") as mock:
        route = mock.post("/plaidstone").mock(return_value=httpx.Response(200))
        await dispatcher._deliver(event)

    assert route.called
    sent = route.calls.last.request.content
    assert sent == _event_to_json(event)
    await dispatcher._http.aclose()
